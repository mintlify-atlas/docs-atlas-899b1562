---
title: 'Create Employee'
api: 'POST /api/employees'
description: 'Create a new employee record in the system'
---

## Authentication

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication
</ParamField>

<Note>
  This endpoint requires **admin** privileges.
</Note>

## Request Body

<ParamField body="personalInfo" type="object" required>
  Personal information about the employee
  
  <Expandable title="personalInfo properties">
    <ParamField body="personalInfo.name" type="string" required>
      Full name of the employee (max 100 characters)
    </ParamField>
    
    <ParamField body="personalInfo.email" type="string" required>
      Email address (must be unique)
    </ParamField>
    
    <ParamField body="personalInfo.identification" type="string" required>
      National ID or employee identification number (must be unique)
    </ParamField>
    
    <ParamField body="personalInfo.phoneNumber" type="string">
      Contact phone number
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="jobInfo" type="object" required>
  Employment and role information
  
  <Expandable title="jobInfo properties">
    <ParamField body="jobInfo.position" type="string" required>
      Employee position. Must be one of:
      - `medico` - Doctor
      - `enfermero` - Nurse
      - `auxiliar_enfermeria` - Nursing assistant
      - `administrativo` - Administrative staff
      - `otro` - Other
    </ParamField>
    
    <ParamField body="jobInfo.department" type="string" required>
      Department assignment. Must be one of:
      - `urgencias` - Emergency
      - `hospitalizacion` - Hospitalization
      - `consulta_externa` - Outpatient
      - `administracion` - Administration
      - `otro` - Other
    </ParamField>
    
    <ParamField body="jobInfo.startDate" type="string" required>
      Employment start date in ISO 8601 format (e.g., `2024-01-15T00:00:00.000Z`)
    </ParamField>
    
    <ParamField body="jobInfo.shift" type="string">
      Work shift. Must be one of:
      - `mañana` - Morning (default)
      - `tarde` - Afternoon
      - `noche` - Night
      - `rotativo` - Rotating
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="hospitalId" type="string" required>
  Identifier of the hospital this employee belongs to
</ParamField>

<ParamField body="wellnessMetrics" type="object">
  Initial wellness metrics (optional, defaults provided)
  
  <Expandable title="wellnessMetrics properties">
    <ParamField body="wellnessMetrics.currentMoodScore" type="number">
      Current mood rating (default: 3)
    </ParamField>
    
    <ParamField body="wellnessMetrics.averageWorkload" type="number">
      Average workload score (default: 3)
    </ParamField>
    
    <ParamField body="wellnessMetrics.teamSupportScore" type="number">
      Team support perception score (default: 3)
    </ParamField>
    
    <ParamField body="wellnessMetrics.satisfactionScore" type="number">
      Job satisfaction score (default: 5)
    </ParamField>
    
    <ParamField body="wellnessMetrics.riskLevel" type="string">
      Burnout risk level. One of: `bajo`, `medio`, `alto` (default: `bajo`)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="gamification" type="object">
  Initial gamification data (optional, defaults provided)
  
  <Expandable title="gamification properties">
    <ParamField body="gamification.totalPoints" type="number">
      Total points earned (default: 0)
    </ParamField>
    
    <ParamField body="gamification.currentStreak" type="number">
      Current consecutive days streak (default: 0)
    </ParamField>
    
    <ParamField body="gamification.maxStreak" type="number">
      Maximum streak achieved (default: 0)
    </ParamField>
    
    <ParamField body="gamification.level" type="number">
      Current level (default: 1)
    </ParamField>
    
    <ParamField body="gamification.badges" type="array">
      Array of badge objects (default: [])
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="isActive" type="boolean">
  Whether the employee is currently active (default: true)
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="message" type="string">
  Success message
</ResponseField>

<ResponseField name="data" type="object">
  Response data
  
  <Expandable title="data properties">
    <ResponseField name="employee" type="object">
      The created employee object with all fields including generated `_id`, `createdAt`, and `updatedAt`
    </ResponseField>
  </Expandable>
</ResponseField>

import requests

url = 'https://api.cuido.com/api/employees'
headers = {
    'Authorization': 'Bearer YOUR_ACCESS_TOKEN',
    'Content-Type': 'application/json'
}
payload = {
    'personalInfo': {
        'name': 'Dr. María García',
        'email': 'maria.garcia@hospital.com',
        'identification': '123456789',
        'phoneNumber': '+57 300 123 4567'
    },
    'jobInfo': {
        'position': 'medico',
        'department': 'urgencias',
        'startDate': '2024-01-15T00:00:00.000Z',
        'shift': 'noche'
    },
    'hospitalId': 'HOSP001'
}

response = requests.post(url, json=payload, headers=headers)
data = response.json()
